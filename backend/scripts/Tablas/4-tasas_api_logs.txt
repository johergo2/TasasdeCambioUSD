-- Table: public.tasas_api_logs

-- DROP TABLE IF EXISTS public.tasas_api_logs;

CREATE TABLE IF NOT EXISTS public.tasas_api_logs
(
    id BIGSERIAL PRIMARY KEY,
    api_key VARCHAR(255) NOT NULL,
    endpoint TEXT NOT NULL,
    ip INET NOT NULL,
    status_code SMALLINT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_tasas_api_logs_api_key ON tasas_api_logs (api_key);
CREATE INDEX idx_tasas_api_logs_created_at ON tasas_api_logs (created_at);
CREATE INDEX idx_tasas_api_logs_status_code ON tasas_api_logs (status_code);
